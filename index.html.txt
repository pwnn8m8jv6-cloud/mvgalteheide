<script>
async function loadData() {
try {
const res = await fetch("https://api.allorigins.win/raw?url=https://www.mvg.de/api/fib/v2/departure?globalId=de:09162:740");
const data = await res.json();

let html = "";
let count = 0;

data.forEach(d => {
if (!d.line.startsWith("U")) return;
if (count >= 5) return;

let min = d.departureTimeCountdown;
let isNow = (min === 0 || min === "0");

if (isNow) min = "JETZT";

html += `
<div class="row">
<div class="top">
<div class="line">${d.line}</div>
<div class="time ${isNow ? "blink" : ""}">${min}</div>
</div>
<div class="dest">${d.destination}</div>
</div>
`;
count++;
});

document.getElementById("app").innerHTML = html;

} catch (e) {
document.getElementById("app").innerHTML = "Keine Daten";
}
}

setInterval(loadData, 30000);
window.onload = loadData;
</script>
